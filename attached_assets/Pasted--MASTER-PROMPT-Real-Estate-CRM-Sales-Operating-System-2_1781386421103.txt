export const users = pgTable('users', {
  id: uuid('id').primaryKey().defaultRandom(),
  name: varchar('name', { length: 100 }).notNull().unique(),
  email: varchar('email', { length: 255 }).notNull().unique(),
  passwordHash: text('password_hash'),
  
  // Role & Status
  role: roleEnum('role').notNull().default('sales'),
  status: statusEnum('status').notNull().default('pending'),
  
  // Profile
  avatarUrl: text('avatar_url'),
  title: varchar('title', { length: 100 }),
  phone: varchar('phone', { length: 20 }),
  facebookUrl: text('facebook_url'),
  instagramUrl: text('instagram_url'),
  bio: text('bio'),
  
  // Relations
  teamLeaderId: uuid('team_leader_id').references(() => users.id),
  
  // Auth
  emailVerified: boolean('email_verified').default(false),
  googleId: text('google_id').unique(),
  facebookId: text('facebook_id').unique(),
  
  // Tracking
  isOnline: boolean('is_online').default(false),
  lastActiveAt: timestamp('last_active_at'),
  approvedBy: uuid('approved_by').references(() => users.id),
  approvedAt: timestamp('approved_at'),
  rejectedAt: timestamp('rejected_at'),
  rejectionReason: text('rejection_reason'),
  
  createdAt: timestamp('created_at').defaultNow().notNull(),
  updatedAt: timestamp('updated_at').defaultNow().notNull(),
}, (table) => ({
  roleIdx: index('users_role_idx').on(table.role),
  statusIdx: index('users_status_idx').on(table.status),
  teamLeaderIdx: index('users_team_leader_idx').on(table.teamLeaderId),
}));

// Enums
export const roleEnum = pgEnum('role', [
  'ceo', 'admin', 'director', 'team_leader', 'sales'
]);

export const statusEnum = pgEnum('status', [
  'pending', 'active', 'rejected', 'suspended'
]);

  id: uuid('id').primaryKey().defaultRandom(),

export const leads = pgTable('leads', {
  id: uuid('id').primaryKey().defaultRandom(),
  
  // Basic Info
  name: varchar('name', { length: 200 }).notNull(),
  phone: varchar('phone', { length: 30 }),
  email: varchar('email', { length: 255 }),
  source: leadSourceEnum('source').default('manual'),
  
  // Status & Pipeline
  status: leadStatusEnum('status').notNull().default('new'),
  
  // Relations
  projectId: uuid('project_id').references(() => projects.id),
  primarySalesId: uuid('primary_sales_id').references(() => users.id),
  
  // Admin controls
  isPhoneHidden: boolean('is_phone_hidden').default(false),
  isEmailHidden: boolean('is_email_hidden').default(false),
  isNotesHidden: boolean('is_notes_hidden').default(false),
  
  // Timing
  deadline: timestamp('deadline'),
  delayedUntil: timestamp('delayed_until'),
  lastActionAt: timestamp('last_action_at'),
  
  // Content
  notes: text('notes'),
  outcome: text('outcome'),
  nextAction: text('next_action'),
  nextActionAt: timestamp('next_action_at'),
  
  // Tracking
  createdBy: uuid('created_by').references(() => users.id),
  createdAt: timestamp('created_at').defaultNow().notNull(),
  updatedAt: timestamp('updated_at').defaultNow().notNull(),
}, (table) => ({
  statusIdx: index('leads_status_idx').on(table.status),
  projectIdx: index('leads_project_idx').on(table.projectId),
  primarySalesIdx: index('leads_primary_sales_idx').on(table.primarySalesId),
  deadlineIdx: index('leads_deadline_idx').on(table.deadline),
  createdAtIdx: index('leads_created_at_idx').on(table.createdAt),

// ALGORITHM 1: Permission Resolver
// packages/permissions/src/resolver.ts
// в”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓ

/**
 * Resolution Priority:
 * 1. User DENY override   в†’ DENY (highest priority)
 * 2. User ALLOW override  в†’ ALLOW
 * 3. Role default         в†’ role permission value
 * 4. Fallback             в†’ DENY
 *
 * Changes to role permissions apply instantly
 * via Redis cache invalidation + realtime broadcast
 */

async function resolvePermission(
  userId: string,
  permissionKey: string
): Promise<boolean> {
  // Step 1: Check Redis cache
  const cacheKey = `perm:${userId}:${permissionKey}`;
  const cached = await redis.get(cacheKey);
  if (cached !== null) return cached === '1';

  // Step 2: Get user with role
  const user = await getUserWithRole(userId);
  if (!user || user.status !== 'active') return false;

  // Step 3: Check user override
  const override = await getUserOverride(userId, permissionKey);
  let result: boolean;

  if (override?.override === 'deny') {
    result = false;
  } else if (override?.override === 'allow') {
    result = true;
  } else {
    // Step 4: Fall back to role permission
    const rolePermission = await getRolePermission(user.role, permissionKey);
    result = rolePermission?.isEnabled ?? false;
  }

  // Step 5: Cache for 5 minutes
  await redis.setex(cacheKey, 300, result ? '1' : '0');
  return result;
}

// Invalidate all users of a role when role permission changes
async function invalidateRolePermissions(role: string): Promise<void> {
  const users = await getUsersByRole(role);
  const pipeline = redis.pipeline();
  for (const user of users) {
    pipeline.del(`perm:${user.id}:*`);
  }
  await pipeline.exec();
  
  // Broadcast real-time permission update
  await realtimeClient.broadcast('permissions', {
    type: 'ROLE_PERMISSION_UPDATED',
    role,
  });
}

// ALGORITHM 2: Lead Distribution
// packages/services/src/leads/distribution.service.ts
// в”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓв”Ѓ

/**
 * Weighted Round Robin Distribution:
 * - Weight based on: current lead count (inverse) + performance score
 * - Sales with fewer leads get higher weight
 * - Top performers get slight priority boost
 * - Blocked sales are excluded
 * - Inactive (offline >24h) get lower weight
 */

async function distributeLead(leadId: string, teamId?: string): Promise<string> {
  const activeSales = await getActiveSalesTeam(teamId);
  
  const weights = await Promise.all(
    activeSales.map(async (sales) => {
      const leadCount = await getCurrentLeadCount(sales.id);
      const perfScore = await getPerformanceScore(sales.id);
      const isOnline = await checkOnlineStatus(sales.id);
      
      // Inverse weight: fewer leads = higher weight
      const loadWeight = Math.max(0, 100 - leadCount * 5);
      const performanceBoost = perfScore * 10;
      const onlineBoost = isOnline ? 20 : 0;
      
      return {
        salesId: sales.id,
        weight: loadWeight + performanceBoost + onlineBoost,
      };
    })
  );
  
  // Weighted random selection
  const totalWeight = weights.reduce((sum, w) => sum + w.weight, 0);
  let random = Math.random() * totalWeight;
  
  for (const { salesId, weight } of weights) {
    random -= weight;
    if (random <= 0) {
      return salesId;
    }
  }
  
  return weights[0].salesId;
}

/**
 * Prevents race conditions in lead assignment
 * Uses Redis SET NX (atomic operation)
 * TTL: 30 seconds (operation timeout)
 */

async function acquireLeadLock(
  leadId: string, 
  userId: string,
  ttlSeconds = 30
): Promise<boolean> {
  const lockKey = `lead_lock:${leadId}`;
  const lockValue = `${userId}:${Date.now()}`;
  
  const result = await redis.set(lockKey, lockValue, {
    nx: true,    // Only set if not exists
    ex: ttlSeconds,
  });
  
  return result === 'OK';
}

async function releaseLeadLock(leadId: string, userId: string): Promise<void> {
  const lockKey = `lead_lock:${leadId}`;
  const currentValue = await redis.get(lockKey);
  
  // Only release if WE own the lock (prevent releasing others' locks)
  if (currentValue?.startsWith(userId)) {
    await redis.del(lockKey);
  }
}

async function maskLeadFields<T extends Record<string, unknown>>(
  lead: T,
  userId: string,
  assignmentHiddenFields: string[]
): Promise<Partial<T>> {
  const canSeePhone = await resolvePermission(userId, 'leads.see_phone');
  const canSeeEmail = await resolvePermission(userId, 'leads.see_email');
  const canSeeNotes = await resolvePermission(userId, 'leads.see_notes');
  
  const masked = { ...lead };
  
  if (!canSeePhone || lead.isPhoneHidden || assignmentHiddenFields.includes('phone')) {
    delete masked.phone;
  }
  if (!canSeeEmail || lead.isEmailHidden || assignmentHiddenFields.includes('email')) {
    delete masked.email;
  }
  if (!canSeeNotes || lead.isNotesHidden || assignmentHiddenFields.includes('notes')) {
    delete masked.notes;
  }

// Client sends heartbeat every 30 seconds
// Server sets Redis key with 35 second TTL
// If key expires в†’ user is offline
// Background job checks DB sync every 5 minutes

async function recordHeartbeat(userId: string): Promise<void> {
  const key = `online:${userId}`;
  await redis.setex(key, 35, Date.now().toString());
  
  // Update DB periodically (not every heartbeat to reduce DB load)
  const lastDbUpdate = await redis.get(`heartbeat_db:${userId}`);
  if (!lastDbUpdate || Date.now() - parseInt(lastDbUpdate) > 300000) {
    await db.update(users)
      .set({ isOnline: true, lastActiveAt: new Date() })
      .where(eq(users.id, userId));
    await redis.setex(`heartbeat_db:${userId}`, 300, Date.now().toString());
  }
}

async function isUserOnline(userId: string): Promise<boolean> {
  const key = `online:${userId}`;
  return (await redis.exists(key)) === 1;
}

  const { userId, dateFrom, dateTo, projectId } = filters;
  
  return await db.execute(sql`
    SELECT
      COUNT(*) FILTER (WHERE status = 'new') as new_leads,
      COUNT(*) FILTER (WHERE status = 'called') as called_leads,
      COUNT(*) FILTER (WHERE status = 'qualified') as qualified_leads,
      COUNT(*) FILTER (WHERE status = 'proposal') as proposal_leads,
      COUNT(*) FILTER (WHERE status = 'negotiation') as negotiation_leads,
      COUNT(*) FILTER (WHERE status = 'won') as won_leads,
      COUNT(*) FILTER (WHERE status = 'lost') as lost_leads,
      COUNT(*) FILTER (
        WHERE created_at::date = CURRENT_DATE
      ) as new_today,
      ROUND(
        COUNT(*) FILTER (WHERE status = 'won')::numeric / 
        NULLIF(COUNT(*), 0) * 100, 
        2
      ) as conversion_rate
    FROM leads
    WHERE
      (${userId}::uuid IS NULL OR primary_sales_id = ${userId}::uuid)
      AND (${dateFrom}::date IS NULL OR created_at >= ${dateFrom}::date)
      AND (${dateTo}::date IS NULL OR created_at <= ${dateTo}::date)
      AND (${projectId}::uuid IS NULL OR project_id = ${projectId}::uuid)
  `);
}

export async function getActivityStats(filters: ActivityStatsFilters) {
  return await db.execute(sql`
    SELECT
      u.id,
      u.name,
      u.avatar_url,
      COUNT(la.id) FILTER (WHERE la.type = 'call') as total_calls,
      COUNT(la.id) FILTER (WHERE la.type = 'meeting') as total_meetings,
      COUNT(la.id) FILTER (WHERE la.type = 'message') as total_messages,
      COUNT(la.id) as total_activities,
      COUNT(DISTINCT la.lead_id) as unique_leads_touched,
      COUNT(l.id) FILTER (WHERE l.status = 'won') as deals_won,
      SUM(c.deal_value) FILTER (WHERE l.status = 'won') as total_revenue,
      ROUND(
        AVG(EXTRACT(EPOCH FROM (la.created_at - l.created_at))/3600)
        , 2
      ) as avg_response_hours
    FROM users u
    LEFT JOIN lead_activities la ON la.user_id = u.id
      AND la.created_at BETWEEN ${filters.dateFrom} AND ${filters.dateTo}
    LEFT JOIN leads l ON la.lead_id = l.id
    LEFT JOIN clients c ON c.lead_id = l.id
    WHERE u.role = 'sales' AND u.status = 'active'
    GROUP BY u.id, u.name, u.avatar_url
    ORDER BY deals_won DESC, total_activities DESC
  `);
}

export async function getTeamStats(teamLeaderId: string) {
  return await db.execute(sql`
    WITH team_members AS (
      SELECT id FROM users WHERE team_leader_id = ${teamLeaderId}
    ),
    member_stats AS (
      SELECT
        u.id,
        u.name,
        COUNT(l.id) as total_leads,
        COUNT(l.id) FILTER (WHERE l.status = 'won') as won,
        COUNT(la.id) as activities,
        BOOL_OR(u.is_online) as is_online
      FROM team_members tm
      JOIN users u ON u.id = tm.id
      LEFT JOIN lead_assignments la2 ON la2.sales_id = u.id
      LEFT JOIN leads l ON l.id = la2.lead_id
      LEFT JOIN lead_activities la ON la.user_id = u.id
      GROUP BY u.id, u.name
    )
    SELECT *, 
      RANK() OVER (ORDER BY won DESC, activities DESC) as rank
    FROM member_stats
  `);
}

export async function generateReport(params: ReportParams) {
  const {
    type,           // 'daily' | 'weekly' | 'monthly' | 'custom'
    dateFrom,
    dateTo,
    userIds,        // filter by specific users
    projectIds,     // filter by projects
    statuses,       // filter by lead statuses
    groupBy,        // 'user' | 'project' | 'status' | 'date'
    format,         // 'json' | 'csv' | 'excel' | 'pdf'
  } = params;
  
  const [leadStats, activityStats, resaleStats, teamStats] = await Promise.all([
    getLeadStats({ dateFrom, dateTo, ...filters }),
    getActivityStats({ dateFrom, dateTo, ...filters }),
    getResaleStats({ dateFrom, dateTo }),
    getTeamStats({ dateFrom, dateTo }),
  ]);
  
  const report = {
    meta: {
      generatedAt: new Date(),
      period: { from: dateFrom, to: dateTo },
      type,
    },
    summary: {
      totalLeads: leadStats.total,
      conversionRate: leadStats.conversion_rate,
      totalRevenue: activityStats.total_revenue,
      topPerformer: activityStats[0],
      avgResponseTime: activityStats.avg_response_hours,
    },
    byEmployee: activityStats,
    byProject: projectStats,
    funnel: leadStats,
    timeline: timelineData,
  };
  
  if (format === 'excel') return exportToExcel(report);
  if (format === 'csv') return exportToCSV(report);
  if (format === 'pdf') return exportToPDF(report);
  return report;
}

// Import bulk leads from Excel/CSV
export async function importLeads(
  file: File,
  assignedBy: string,
  defaultProjectId?: string
): Promise<ImportResult> {
  const workbook = read(await file.arrayBuffer());
  const sheet = workbook.Sheets[workbook.SheetNames[0]];
  const rows = utils.sheet_to_json(sheet);
  
  const results = {
    success: 0,
    failed: 0,
    errors: [] as ImportError[],
  };
  
  // Validate each row with Zod
  for (const [index, row] of rows.entries()) {
    const parsed = BulkLeadSchema.safeParse(row);
    
    if (!parsed.success) {
      results.failed++;
      results.errors.push({
        row: index + 2,
        field: parsed.error.issues[0].path[0],
        message: parsed.error.issues[0].message,
      });
      continue;
    }
    
    await db.insert(leads).values({
      ...parsed.data,
      projectId: parsed.data.projectId ?? defaultProjectId,
      source: 'import',
      createdBy: assignedBy,
    });
    
    results.success++;
  }
  
  await auditLog({
    userId: assignedBy,
    action: 'BULK_IMPORT_LEADS',
    entityType: 'leads',
    metadata: { total: rows.length, ...results },
  });
  
  return results;
}

// Client hook (React)
export function useRealtimeChannel(
  channel: string,
  handler: (event: RealtimeEvent) => void
): void {
  useEffect(() => {
    const subscription = supabaseClient
      .channel(channel)
      .on('broadcast', { event: '*' }, ({ payload }) => {
        handler(payload as RealtimeEvent);
      })
      .subscribe();
    
    return () => { supabaseClient.removeChannel(subscription); };
  }, [channel]);
}

// apps/web/middleware.ts

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl;
  
  // 1. i18n routing
  const locale = detectLocale(request);
  
  // 2. Public routes bypass
  if (isPublicRoute(pathname)) {
    return intlMiddleware(request);
  }
  
  // 3. Auth check
  const session = await getSession(request);
  if (!session) {
    return redirectToLogin(request, locale);
  }
  
  // 4. Status check
  if (session.user.status !== 'active') {
    return redirectToPending(request, locale);
  }
  
  // 5. Rate limiting
  const rateLimitResult = await rateLimit(session.user.id, pathname);
  if (!rateLimitResult.success) {
    return new NextResponse('Too Many Requests', { status: 429 });
  }
  
  // 6. Attach user to headers for Route Handlers
  const requestHeaders = new Headers(request.headers);
  requestHeaders.set('x-user-id', session.user.id);
  requestHeaders.set('x-user-role', session.user.role);
  
  return intlMiddleware(request, { requestHeaders });
}

// Route Handler middleware pattern
export function withPermission(
  permissionKey: string,
  handler: (req: Request, ctx: RouteContext) => Promise<Response>
) {
  return async (req: Request, ctx: RouteContext) => {
    const userId = req.headers.get('x-user-id');
    if (!userId) return unauthorized();
    
    const hasPermission = await resolvePermission(userId, permissionKey);
    if (!hasPermission) return forbidden();
    
    // Log the API call to audit
    const response = await handler(req, ctx);
    
    return response;
  };
}
